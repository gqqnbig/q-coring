country_event = {
	id = cor_sta.1
	fire_only_once = no
	mean_time_to_happen = { days = 1 }
	hidden = no
	
	title = cor_sta.1.t
	desc = cor_sta.1.d
	
	trigger = {
		any_state = {
			AND = {
				is_controlled_by = FROM
				NOT = { is_core_of = FROM }
				if = {
					LIMIT = { 
						NOT = { is_owned_by = FROM }
					}
					all_neighbor_state = {
						is_controlled_by = FROM
					}
				}
				OR = {
					any_neighbor_state = {
						OR = {
							is_core_of = FROM
							has_state_flag = integrated_by_@FROM
						}
					}
					AND = {
						is_coastal = yes
						FROM = {
							any_owned_state = {
								AND = {
									is_core_of = FROM
									is_coastal = yes
								}
							}
						}
					}
				}
				AND = {
					NOT = { has_state_flag = integrated_by_@FROM }
					NOT = { has_state_flag = fast_integration_@FROM }
					NOT = { has_state_flag = normal_integration_@FROM }
					NOT = { has_state_flag = slow_integration_@FROM }
				}
			}
		}
	}
	
	option = { #fast integration
		trigger = {
			AND = {
				has_political_power > 29
				OR = {
					is_ai = no
					has_political_power > 299
				}
			}
		}
		name = cor_sta.1.a
		random_state = {
			limit = {
				AND = {
					is_controlled_by = FROM
					NOT = { is_core_of = FROM }
					if = {
						LIMIT = { 
							NOT = { is_owned_by = FROM }
						}
						all_neighbor_state = {
							is_controlled_by = FROM
						}
					}
					OR = {
						any_neighbor_state = {
							OR = {
								is_core_of = FROM
								has_state_flag = integrated_by_@FROM
							}
						}

							is_coastal = yes
							FROM = {
								any_owned_state = {
									AND = {
										is_core_of = FROM
										is_coastal = yes
									}
								}
							}
						}
					}
					AND = {
						NOT = { has_state_flag = integrated_by_@FROM }
						NOT = { has_state_flag = fast_integration_@FROM }
						NOT = { has_state_flag = normal_integration_@FROM }
						NOT = { has_state_flag = slow_integration_@FROM }

				}
			}
			set_state_flag = fast_integration_@FROM

	option = { #normal integration

				}
			}
		}
		name = cor_sta.1.b
		random_state = {
			limit = {
				AND = {
					is_controlled_by = FROM
					NOT = { is_core_of = FROM }
					AND = {
						NOT = { has_state_flag = integrated_by_@FROM }
						NOT = { has_state_flag = fast_integration_@FROM }
						NOT = { has_state_flag = normal_integration_@FROM }
						NOT = { has_state_flag = slow_integration_@FROM }
					}

							NOT = { is_owned_by = FROM }
						}

							is_controlled_by = FROM
						}
					}
					OR = {

								is_core_of = FROM

					}
